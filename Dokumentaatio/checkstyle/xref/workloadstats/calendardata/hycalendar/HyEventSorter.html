29          this.ug = new UidGenerator("uidGen");

33      /**
34       * Compile new Course event-wrapper
35       *
36       * @param oldSummary summary from the University calendar
37       * @param newSummary new summary from the user
38       * @param courseEvents List of events identified as the main lectures of
39       * this course, used as a source of information
40       * @return new Course event-wrapper
41       */
42      public Course getNewCourse(String oldSummary, String newSummary, List<Event> courseEvents) {
43          if (courseEvents == null) {
44              return null;
45          }
46          PropertyList props = new PropertyList();
47          props.add(new DtStart(getStartDate(courseEvents)));
48          props.add(new DtEnd(getEndDate(courseEvents)));
49          props.add(new DtStamp());
50          props.add(new Summary(newSummary));
51          props.add(ug.generateUid());
52          props.add(new Categories(oldSummary));
53          props.add(new Categories("COURSE"));
54          props.add(VEVENT_TENTATIVE);
55  
56          return new Course(props);
57      }
58  
59      /**
60       * Get earliest value in DTStart in a list of events
61       *
62       * @param events List of events used as a time frame
63       * @return earliest found DTstart date
64       */
65      public Date getStartDate(List<Event> events) {
66          if (events == null) {
67              return null;
68          }
69          Date start = events.get(0).getStartDate().getDate();
70          for (Event event : events) {
71              if (event.getStartDate().getDate().before(start)) {
72                  start = event.getStartDate().getDate();
73              }
74          }
75          return start;
76      }
77  
78      /**
79       * Get the last value in DTEnd in a list of events
80       *
81       * @param events List of events used as a time frame
82       * @return last found DTEnd date
83       */
84      public Date getEndDate(List<Event> events) {
85          if (events == null) {
86              return null;
87          }
88          Date end = events.get(0).getStartDate().getDate();
89          for (Event event : events) {
90              if (event.getEndDate().getDate().after(end)) {
91                  end = event.getEndDate().getDate();
92              }
93          }
94          return end;
95      }

97      /**
98       * Turn a list of University calendar events identified as lectures to a
99       * list of Lecture objects in Event-wrappers
100      *
101      * @param newSummary New summary provided by the user
102      * @param courseEvents List of events from University calendar that were
103      * identified as lectures, but are still generic Event objects
104      * @return List of Lecture-objects
105      */
106     public List<Event> getNewLectures(String newSummary, List<Event> courseEvents) {
107         if (courseEvents == null) {
108             return null;
109         }
110         List<Event> lectures = new ArrayList<>();
111         for (Event courseEvent : courseEvents) {
112             Lecture newLecture = new Lecture(courseEvent);
113             newLecture.getSummary().setValue(newSummary + " Luento");
114             newLecture.getProperties().add(new Categories("LECTURE"));
115             newLecture.getProperties().add(VEVENT_TENTATIVE);
116             newLecture.getProperties().add(new DtStamp());
117             lectures.add(newLecture);
118         }
119 
120         return lectures;
121     }
122     /**
123      * Turn a list of University calendar events identified as exercises to a
124      * list of Exercise objects in Event-wrappers
125      * @param newSummary New summary provided by the user
126      * @param exerciseEvents List of events from University calendar that were
127      * identified as exercises, but are still generic Event objects
128      * @return List of Exercise-objects
129      */
130     List<Event> getNewExercises(String newSummary, List<Event> exerciseEvents) {
131         if (exerciseEvents == null) {
132             return null;
133         }
134         List<Event> exercises = new ArrayList<>();
135         for (Event exerciseEvent : exerciseEvents) {
136             Exercise newExercise = new Exercise(exerciseEvent);
137             newExercise.getSummary().setValue(newSummary + " Harkka");
138             newExercise.getProperties().add(new Categories("EXERCISE"));
139             newExercise.getProperties().add(VEVENT_TENTATIVE);
140             newExercise.getProperties().add(new DtStamp());
141             exercises.add(newExercise);
142         }
143         return exercises;
144     }
