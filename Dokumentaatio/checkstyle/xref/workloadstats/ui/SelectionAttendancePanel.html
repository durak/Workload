1   package workloadstats.ui;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import javax.swing.ButtonGroup;
6   import javax.swing.JPanel;
7   import javax.swing.JRadioButton;
8   import workloadstats.domain.model.Event;
9   import workloadstats.utils.Ac;
10  
11  /**
12   * Radiobuttonpanel for event attendance status changes
13   * @author Ilkka
14   */
15  public class SelectionAttendancePanel extends JPanel {
16      
17      private Ac[] buttonIds;
18      private List<JRadioButton> buttons;
19      private ButtonGroup group;
20      private SelectionAttendanceListener esl;
21  
22      public SelectionAttendancePanel(Ac[] buttonIds, String title, SelectionAttendanceListener esl) {
23          this.setBorder(javax.swing.BorderFactory.createTitledBorder(title));
24          this.buttonIds = buttonIds;
25          this.esl = esl;
26          this.buttons = new ArrayList<>();
27          this.group = new ButtonGroup();
28          initComponents();
29      }
30  
31      private void initComponents() {        
32          for (Ac buttonId : buttonIds) {            
33              JRadioButton jrb = new JRadioButton(buttonId.getDescr());
34              jrb.setActionCommand(buttonId.name());
35              jrb.setEnabled(false);
36              jrb.addActionListener(esl);
37              buttons.add(jrb);
38              group.add(jrb);
39              this.add(jrb);
40          }
41  
42      }
43      
44      public void resetGroup(Event selected) {
45          for (JRadioButton button : buttons) {
46              if (selected == null) {
47                  button.setEnabled(false);
48              } else {
49                  button.setSelected(button.getActionCommand().equals(selected.getEventStatus()));
50                  button.setEnabled(true);
51              }
52          }
53      }
54  }
