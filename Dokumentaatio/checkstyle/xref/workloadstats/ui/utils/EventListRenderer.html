1   package workloadstats.ui.utils;
2   
3   import java.awt.Component;
4   import javax.swing.JLabel;
5   import javax.swing.JList;
6   import javax.swing.ListCellRenderer;
7   import workloadstats.domain.model.Event;
8   
9   /**
10   * Custom rendered for a JList of events
11   * @author Ilkka
12   */
13  public class EventListRenderer extends JLabel implements ListCellRenderer<Event> {
14  
15      public EventListRenderer() {
16          setOpaque(true);
17      }
18  
19      @Override
20      public Component getListCellRendererComponent(JList<? extends Event> list, Event event, int index,
21              boolean isSelected, boolean cellHasFocus) {
22          
23          String eventName = event.getEventName();
24          String eventDate = event.getStartDateString();
25          setText(eventDate + " " + eventName);
26          if (isSelected) {
27              setBackground(list.getSelectionBackground());
28              setForeground(list.getSelectionForeground());
29          } else {
30              setBackground(list.getBackground());
31              setForeground(list.getForeground());
32          }
33  
34          return this;
35      }
36  
37  }
