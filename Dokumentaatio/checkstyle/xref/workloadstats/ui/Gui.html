1   package workloadstats.ui;
2   
3   import java.awt.BorderLayout;
4   import java.awt.Container;
5   import java.awt.Dimension;
6   import javax.swing.BoxLayout;
7   import javax.swing.JFrame;
8   import javax.swing.JList;
9   import javax.swing.JPanel;
10  import javax.swing.WindowConstants;
11  import workloadstats.calendardata.MyCalendarControl;
12  import workloadstats.utils.Ac;
13  
14  /**
15   * Main GUI builder
16   * @author Ilkka
17   */
18  public class Gui implements Runnable {
19  
20      private MyCalendarControl myCalendarControl;
21      private JFrame frame;
22  
23      public Gui(MyCalendarControl myCalendarControl) {
24          this.myCalendarControl = myCalendarControl;
25      }
26  
27      @Override
28      public void run() {
29          frame = new JFrame("WorkloadStats");
30          frame.setPreferredSize(new Dimension(1200, 800));
31          frame.setResizable(false);
32          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
33  
34          initComponent(frame.getContentPane());
35  
36          frame.pack();
37          frame.setVisible(true);
38      }
39  
40      private void initComponent(Container container) {
41  
42          // Initialize Data model       
43          CourseListModel courseListModel = new CourseListModel(myCalendarControl.getCourses());
44          EventListModel eventListModel = new EventListModel(courseListModel);
45  
46          // Initialize Course and Event lists with Data models
47          // datalistener not needed in current setup
48          // clm.addListDataListener(elm);
49          JList courseList = new JList(courseListModel);
50          courseList.setName("courselist");
51          JList eventList = new JList(eventListModel);
52          eventList.setName("eventlist");
53  
54          // Initialize CourseListPanel and EventListPanel
55          CourseListPanel courseListPanel = new CourseListPanel(courseList);
56          EventListPanel middleEventListPanel = new EventListPanel(eventList);
57  
58          // Initialize Main menu
59          Ac[] menuButtons = {Ac.NEWCALENDAR, Ac.LOADCALENDAR, Ac.IMPORTCALENDAR, Ac.SAVECALENDAR};
60          ButtonsPanel mainMenu = new ButtonsPanel(menuButtons, "Main menu");
61          mainMenu.setMaximumSize(new Dimension(400, 20));
62  
63          // Initialize Calendar buttons
64          Ac[] eventButtons = {Ac.NEWCOURSE, Ac.DELETECOURSE, Ac.NEWEVENT, Ac.DELETE_EVENT};
65          ButtonsPanel calendarActionPanel = new ButtonsPanel(eventButtons, "Toiminnot");
66          calendarActionPanel.setMaximumSize(new Dimension(400, 20));
67  
68          // Initialize Selection Statistics & Selection Information panels
69          SelectionStatsPanel selectionStatsPanel = new SelectionStatsPanel(eventListModel);
70          SelectionInformationPanel selectionInfoPanel = new SelectionInformationPanel();
71  
72          // Initialize Calendar and Main Menu listeners 
73          CalendarActionListener calActListener = new CalendarActionListener(container, courseList, eventListModel, calendarActionPanel.getButtons());
74          MainMenuActionListener mainMenuLstr = new MainMenuActionListener(container, myCalendarControl, courseListModel, courseList);
75  
76          /*
77           *  Add Listeners:
78           *  CalendarActionListener
79           *   -> Course list (ListSelectionListener)
80           *   -> Event list (ListSelectionListener)
81           *   -> Calendar buttons (ActionListener)
82           */
83          courseList.addListSelectionListener(calActListener);
84          eventList.addListSelectionListener(calActListener);
85          calendarActionPanel.addListener(calActListener);
86  
87          /*
88           *  MainMenuActionListener
89           *   -> mainMenu         
90           */
91          mainMenu.addListener(mainMenuLstr);
92          /*
93           *  SelectionStatsPanel
94           *   -> EventList (ListSelectionListener)
95           */
96          eventList.addListSelectionListener(selectionStatsPanel);
97          /*
98           *  SelectionInformationPanel
99           *   -> EventList (ListSelectionListener)
100          */
101         eventList.addListSelectionListener(selectionInfoPanel);
102 
103         /*
104          *  EventListPanel
105          *   -> EventListModel (ListDataListener)
106          */
107         eventListModel.addListDataListener(middleEventListPanel);
108         /*
109          *  EventDataModel
110          *   -> CourseList (ListSelectionListener)
111          */
112         courseList.addListSelectionListener(eventListModel);
113 
114         // Construct user interface with initialized parts
115         JPanel west = new JPanel();
116         west.setLayout(new BoxLayout(west, BoxLayout.Y_AXIS));
117         west.add(mainMenu);
118         west.add(courseListPanel);
119         west.add(calendarActionPanel);
120 
121         JPanel east = new JPanel();
122         east.setLayout(new BoxLayout(east, BoxLayout.Y_AXIS));
123         east.add(selectionInfoPanel);
124         east.add(selectionStatsPanel);
125 
126         Dimension eastDimension = new Dimension(400, 800);
127         Dimension westDimension = new Dimension(300, 800);
128         west.setPreferredSize(westDimension);
129         east.setPreferredSize(eastDimension);
130         middleEventListPanel.setSize(middleEventListPanel.getPreferredSize());
131 
132         container.add(west, BorderLayout.WEST);
133         container.add(middleEventListPanel, BorderLayout.CENTER);
134         container.add(east, BorderLayout.EAST);
135     }
136 
137 }
